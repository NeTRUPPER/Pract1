Команды для работы с программой

1. INSERT - Вставка данных

Синтаксис:

INSERT INTO <имя_таблицы> VALUES ('значение1', 'значение2', 'значение3', ...)


Примеры:

INSERT INTO table1 VALUES ('значение1', 'значение2', 'значение3', 'значение4')
INSERT INTO table2 VALUES ('данные1', 'данные2')

2. DELETE - Удаление данных

Синтаксис:

DELETE FROM <имя_таблицы>
WHERE <имя_таблицы>.<имя_колонки> = 'значение'


Примеры:

DELETE FROM table1
WHERE table1.column1 = 'значение1'


DELETE FROM table2
WHERE table2.column2 = 'данные2'


3. SELECT - Выборка данных

Синтаксис (базовый, без WHERE):

SELECT <таблица1>.<колонка1> <таблица2>.<колонка2>
FROM <таблица1>, <таблица2>


Синтаксис (с условием WHERE):

SELECT <таблица1>.<колонка1> <таблица2>.<колонка2>
FROM <таблица1>, <таблица2>
WHERE <таблица1>.<колонка1> = 'значение' [AND|OR <таблица2>.<колонка2> = 'значение']

Примеры:

Простой SELECT без условий:

SELECT table1.column1 table2.column2
FROM table1, table2


SELECT с одним условием:

SELECT table1.column1 table2.column2
FROM table1, table2
WHERE table1.column1 = 'значение1'


SELECT с условием AND:

SELECT table1.column1 table2.column2
FROM table1, table2
WHERE table1.column1 = 'значение1' AND table2.column2 = 'значение2'


SELECT с условием OR:

SELECT table1.column1 table2.column2
FROM table1, table2
WHERE table1.column1 = 'значение1' OR table2.column2 = 'значение2'


SELECT с сравнением колонок:

SELECT table1.column1 table2.column2
FROM table1, table2
WHERE table1.column1 = table2.column1
